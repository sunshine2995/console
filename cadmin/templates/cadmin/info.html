GITLAB_SERVICE_API：


GITLAB_SERVICE_API = {
    'url': 'http://example.example.com/',
    'admin_user': 'user',
    'admin_email': 'app@example.com',
    'admin_password': 'example_password',
    'apitype': 'gitlab service',
    'hook_url': 'https://user.goodrain.com/service/gitlabhook/',
}

GITHUB_SERVICE_API：


GITHUB_SERVICE_API = {
